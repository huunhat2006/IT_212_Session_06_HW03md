package com.speedycart.orderservice.service;

import com.speedycart.orderservice.exception.InvalidOrderException;
import com.speedycart.orderservice.exception.OutOfStockException;
import com.speedycart.orderservice.exception.PaymentFailedException;
import com.speedycart.orderservice.exception.ProductNotFoundException;
import com.speedycart.orderservice.model.Order;
import com.speedycart.orderservice.model.OrderItem;
import com.speedycart.orderservice.model.Product;
import com.speedycart.orderservice.repository.ProductRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.math.BigDecimal;
import java.util.List;

@Service
public class OrderService {

    private static final Logger log = LoggerFactory.getLogger(OrderService.class);

    private final ProductRepository productRepository;
    private final PaymentGateway paymentGateway;

    public OrderService(ProductRepository productRepository, PaymentGateway paymentGateway) {
        this.productRepository = productRepository;
        this.paymentGateway = paymentGateway;
    }

    /**
     * Xử lý quy trình đặt hàng cốt lõi một cách bền vững (Robust).
     * Sử dụng @Transactional để đảm bảo nếu thanh toán lỗi thì tồn kho sẽ được rollback.
     */
    @Transactional(rollbackFor = Exception.class)
    public void processOrderOrder(Order order) {
        log.info("Bắt đầu xử lý đơn hàng: {}", order != null ? order.getId() : "NULL");

        // 1. Kiểm tra tính hợp lệ của đơn hàng (Defensive Programming)
        validateOrder(order);

        // 2. Kiểm tra tồn kho và khấu trừ (Deduct Inventory)
        deductInventory(order.getItems());

        // 3. Xử lý thanh toán qua cổng thanh toán quốc tế/nội địa
        processPayment(order);

        log.info("Xử lý đơn hàng {} thành công.", order.getId());
    }

    /**
     * Xác thực cấu trúc đơn hàng trước khi xử lý sâu hơn vào DB.
     */
    private void validateOrder(Order order) {
        if (order == null) {
            throw new InvalidOrderException("Đơn hàng không được phép rỗng (null).");
        }
        
        List<OrderItem> items = order.getItems();
        if (items == null || items.isEmpty()) {
            throw new InvalidOrderException("Đơn hàng phải chứa ít nhất một sản phẩm.");
        }
    }

    /**
     * Kiểm tra trạng thái sản phẩm, validate số lượng tồn kho và tiến hành trừ kho.
     */
    private void deductInventory(List<OrderItem> items) {
        for (OrderItem item : items) {
            String productId = item.getProductId();
            int requestedQuantity = item.getQuantity();

            // Tìm kiếm sản phẩm, tránh dùng Optional.orElse(null) để không gây giấu lỗi NullPointerException
            Product product = productRepository.findById(productId)
                    .orElseThrow(() -> {
                        log.error("Không tìm thấy sản phẩm có ID: {}", productId);
                        return new ProductNotFoundException("Sản phẩm không tồn tại: " + productId);
                    });

            // Kiểm tra tính hợp lệ của số lượng tồn kho
            int availableStock = product.getStock();
            if (availableStock <= 0 || requestedQuantity > availableStock) {
                log.warn("Sản phẩm {} hết hàng hoặc không đủ. Tồn kho: {}, Yêu cầu: {}", 
                        productId, availableStock, requestedQuantity);
                throw new OutOfStockException(
                        String.format("Sản phẩm %s đã hết hàng hoặc không đủ số lượng cung cấp (Tồn kho: %d).", 
                                product.getName(), availableStock)
                );
            }

            // Tiến hành cập nhật giảm số lượng tồn kho
            product.setStock(availableStock - requestedQuantity);
            productRepository.save(product);
            log.debug("Đã trừ kho sản phẩm {}. Số lượng còn lại: {}", productId, product.getStock());
        }
    }

    /**
     * Đóng gói phần tương tác với bên thứ 3 (Payment Gateway), bọc lỗi ngoại lệ phòng vệ.
     */
    private void processPayment(Order order) {
        BigDecimal amount = order.getTotalAmount();
        try {
            log.info("Yêu cầu thanh toán số tiền {} cho đơn hàng {}", amount, order.getId());
            
            // Gọi sang service của bên thứ 3
            paymentGateway.charge(order.getUserId(), amount);
            
        } catch (Exception e) {
            // Mẫu thiết kế Robustness: Bọc toàn bộ Unchecked/Checked Exception từ bên thứ 3 thành lỗi nghiệp vụ của mình
            log.error("Lỗi nghiêm trọng khi tương tác với Payment Gateway của đơn hàng {}: {}", 
                    order.getId(), e.getMessage(), e);
            
            throw new PaymentFailedException("Thanh toán thất bại do lỗi cổng thanh toán: " + e.getMessage(), e);
        }
    }
}

interface PaymentGateway {
    void charge(String userId, BigDecimal amount) throws Exception;
}
Các điểm cải tiến cốt lõi về Robustness & Clean Code:
Ngăn chặn triệt để NullPointerException: Bằng cách kiểm tra tuần tự từ gốc order != null cho tới ngọn order.getItems().isEmpty(). Việc check items == null trước khi gọi .isEmpty() giúp hệ thống không bao giờ bị crash bất ngờ.

Loại bỏ Optional.orElse(null): Việc sử dụng cấu trúc .orElseThrow(() -> new ProductNotFoundException(...)) giúp luồng code đi đúng tư duy Fail-Fast (Thất bại sớm), code không bị "ô nhiễm" bởi các câu lệnh check if (product == null) ở phía sau.

Phòng thủ lỗi tích hợp bên thứ ba (Third-party Fault Tolerance): Các Cổng thanh toán (Payment Gateway) rất dễ bị timeout, lỗi mạng hoặc lỗi API. Việc bọc toàn bộ khối paymentGateway.charge() trong try-catch (Exception e) đảm bảo rằng dù thư viện của họ có ném ra bất kỳ loại Exception kỳ dị nào, hệ thống của ta vẫn bắt được (catch) và chuyển đổi (wrap) thành PaymentFailedException rõ ràng.

Đảm bảo tính toàn vẹn dữ liệu (Atomicity): Thêm annotation @Transactional(rollbackFor = Exception.class). Nếu việc thanh toán ở bước 3 thất bại (ném ra PaymentFailedException), toàn bộ tiến trình trừ kho ở bước 2 sẽ được database Rollback tự động về trạng thái ban đầu, tránh việc user bị trừ kho vô lý khi chưa trả tiền.

1. Vì sao @Transactional đảm bảo tính Atomicity (Tính nguyên tử)?
Trong hệ thống tài chính và TMĐT, Atomicity (Tính nguyên tử) thuộc quy tắc ACID yêu cầu: "Hoặc tất cả các thao tác trong giao dịch đều thành công, hoặc không có thao tác nào được ghi nhận (All or Nothing)".

Khi ta cấu hình @Transactional(rollbackFor = Exception.class) bọc quanh phương thức processOrder:

Thread Bound Session: Spring sẽ mở một Database Transaction tương ứng với Thread đang xử lý.

Intercepting & Monitoring: Spring sử dụng cơ chế AOP (Aspect-Oriented Programming) để giám sát phương thức này.

Automatic Rollback: Nếu có bất kỳ ngoại lệ nào xảy ra (kể cả Checked hay Unchecked Exception nhờ cấu hình rollbackFor = Exception.class), Spring sẽ lập tức phát lệnh ROLLBACK xuống Database.

Kết quả: Toàn bộ các câu lệnh cập nhật trước đó (ví dụ: đã trừ kho của 2 sản phẩm trước khi đến bước thanh toán bị lỗi) sẽ bị hủy bỏ hoàn toàn. Dữ liệu kho được khôi phục về trạng thái như chưa hề có cuộc giao dịch nào, ngăn chặn triệt để tình trạng lỗi mất cân bằng dữ liệu (Ví dụ: Khách bị trừ kho nhưng thanh toán thất bại).

package com.speedycart.orderservice.service;

import com.speedycart.orderservice.exception.*;
import com.speedycart.orderservice.model.Order;
import com.speedycart.orderservice.model.OrderItem;
import com.speedycart.orderservice.model.Product;
import com.speedycart.orderservice.repository.OrderRepository;
import com.speedycart.orderservice.repository.ProductRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.math.BigDecimal;
import java.util.List;

@Slf4j
@Service
@RequiredArgsConstructor
public class OrderService {

    private final ProductRepository productRepository;
    private final OrderRepository orderRepository;
    private final PaymentGateway paymentGateway;

    /**
     * Quy trình xử lý đơn hàng Production-ready.
     * Đảm bảo tính Nguyên tử (Atomicity) tuyệt đối cho toàn bộ chuỗi thao tác ghi DB.
     */
    @Transactional(rollbackFor = Exception.class)
    public void processOrder(Order order) {
        // Thực hiện validate cấu trúc dữ liệu đầu vào trước khi log sâu
        validateOrderStructure(order);

        String orderId = order.getId();
        String customerId = order.getCustomerId();
        BigDecimal amount = order.getTotalAmount();

        log.info("[START] Bắt đầu xử lý đơn hàng. OrderId: {}, CustomerId: {}, Amount: {}", orderId, customerId, amount);

        try {
            // 1. Kiểm tra kho và Trừ kho
            log.info("[INVENTORY] Kiểm tra kho cho đơn hàng. OrderId: {}, CustomerId: {}", orderId, customerId);
            deductInventory(order.getItems(), orderId, customerId, amount);

            // 2. Gọi cổng thanh toán
            log.info("[PAYMENT] Gọi payment gateway xử lý giao dịch. OrderId: {}, CustomerId: {}, Amount: {}", orderId, customerId, amount);
            processPayment(order, orderId, customerId, amount);

            // 3. Lưu thông tin đơn hàng vào Database
            log.info("[SAVE] Lưu đơn hàng vào database trạng thái thành công. OrderId: {}, CustomerId: {}", orderId, customerId);
            order.setStatus("SUCCESS");
            orderRepository.save(order);

            log.info("[COMPLETED] Hoàn tất xử lý đơn hàng thành công. OrderId: {}, CustomerId: {}", orderId, customerId);

        } catch (OutOfStockException e) {
            log.error("[ERROR] Hết hàng khi xử lý đơn hàng. OrderId: {}, CustomerId: {}, Amount: {}. Lý do: {}", 
                    orderId, customerId, amount, e.getMessage());
            throw e;
        } catch (PaymentFailedException e) {
            log.error("[ERROR] Payment fail - Thanh toán thất bại. OrderId: {}, CustomerId: {}, Amount: {}. Lý do: {}", 
                    orderId, customerId, amount, e.getMessage());
            throw e;
        } catch (OrderException e) {
            log.error("[ERROR] Lỗi nghiệp vụ đơn hàng. OrderId: {}, CustomerId: {}, Amount: {}. Lý do: {}", 
                    orderId, customerId, amount, e.getMessage());
            throw e;
        } catch (Exception e) {
            log.error("[ERROR] Lỗi không mong muốn xảy ra hệ thống. OrderId: {}, CustomerId: {}, Amount: {}. Chi tiết: {}", 
                    orderId, customerId, amount, e.getMessage(), e);
            // Ném ngoại lệ runtime để kích hoạt cơ chế Rollback của Spring @Transactional
            throw new OrderException("Hệ thống gặp sự cố không mong muốn khi xử lý đơn hàng: " + orderId, e);
        }
    }

    private void validateOrderStructure(Order order) {
        if (order == null) {
            throw new InvalidOrderException("Dữ liệu đơn hàng đầu vào bị trống (null).");
        }
        if (order.getItems() == null || order.getItems().isEmpty()) {
            throw new InvalidOrderException("Đơn hàng không hợp lệ do không chứa sản phẩm nào.");
        }
    }

    private void deductInventory(List<OrderItem> items, String orderId, String customerId, BigDecimal amount) {
        for (OrderItem item : items) {
            String productId = item.getProductId();
            int requestedQty = item.getQuantity();

            Product product = productRepository.findById(productId)
                    .orElseThrow(() -> new ProductNotFoundException("Không tìm thấy sản phẩm trên hệ thống: " + productId));

            int currentStock = product.getStock();
            if (currentStock <= 0 || requestedQty > currentStock) {
                throw new OutOfStockException(
                        String.format("Sản phẩm %s không đủ tồn kho (Yêu cầu: %d, Hiện có: %d)", 
                                product.getName(), requestedQty, currentStock)
                );
            }

            log.info("[INVENTORY_DEDUCT] Trừ kho sản phẩm: {} | Số lượng: {}. OrderId: {}, CustomerId: {}", 
                    productId, requestedQty, orderId, customerId);
            
            product.setStock(currentStock - requestedQty);
            productRepository.save(product);
        }
    }

    private void processPayment(Order order, String orderId, String customerId, BigDecimal amount) {
        try {
            paymentGateway.charge(customerId, amount);
        } catch (Exception e) {
            throw new PaymentFailedException("Cổng thanh toán phản hồi lỗi hoặc timeout.", e);
        }
    }
}

1. Cơ chế @Transactional Rollback Inventory trong thực tế
Trong môi trường thực tế, khi paymentGateway.charge() ném ra RuntimeException, quá trình rollback dữ liệu tồn kho (Inventory) diễn ra như sau:

Proxy Interception: OrderPlacementService thực chất là một Spring Bean được bọc bởi một Dynamic Proxy. Khi phương thức placeOrder() được gọi, Proxy sẽ mở một Database Connection và bắt đầu một Transaction (bằng cách thiết lập Connection.setAutoCommit(false)).

Database Session & Locks: Trong quá trình thực hiện hàm deductInventory(), câu lệnh productRepository.save(product) sẽ sinh ra lệnh SQL UPDATE products SET stock = ... WHERE id = .... Lúc này, hệ quản trị cơ sở dữ liệu (RDBMS) sẽ giữ các bản ghi này trong trạng thái Undo Log hoặc Write-Ahead Logging (WAL) và đặt một hàng khóa (Row Lock) để ngăn các transaction khác sửa đổi. Dữ liệu mới này chưa hề được ghi cố định xuống đĩa cứng (Disk).

Exception Triggered: Khi luồng xử lý đi tới processPayment() và gặp RuntimeException, ngoại lệ này sẽ được lan truyền (bubble up) ngược lại cho lớp Spring Proxy.

Rollback Command: Vì ta cấu hình rollbackFor = Exception.class, Spring Proxy ngay lập tức bắt lấy ngoại lệ này và phát lệnh Connection.rollback() xuống cơ sở dữ liệu trước khi ném tiếp Exception ra ngoài.

Khôi phục trạng thái: Hệ quản trị CSDL nhận lệnh rollback sẽ hủy bỏ toàn bộ các thay đổi tạm thời trong bản ghi Undo Log của phiên làm việc đó và giải phóng Row Lock. Do đó, số lượng tồn kho của sản phẩm quay trở về giá trị ban đầu như chưa từng có biến động.

package com.speedycart.orderservice.service;

import com.speedycart.orderservice.dto.OrderPlacementResult;
import com.speedycart.orderservice.exception.*;
import com.speedycart.orderservice.model.Order;
import com.speedycart.orderservice.model.OrderItem;
import com.speedycart.orderservice.model.Product;
import com.speedycart.orderservice.repository.OrderRepository;
import com.speedycart.orderservice.repository.ProductRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.util.List;

@Slf4j
@Service
@RequiredArgsConstructor
public class OrderPlacementService {

    private final ProductRepository productRepository;
    private final OrderRepository orderRepository;
    private final PaymentGateway paymentGateway;

    /**
     * Tiến hành đặt hàng và trả về kết quả chi tiết dưới dạng DTO.
     */
    @Transactional(rollbackFor = Exception.class)
    public OrderPlacementResult placeOrder(Order order) {
        validateOrderStructure(order);

        String orderId = order.getId();
        String customerId = order.getCustomerId();
        BigDecimal amount = order.getTotalAmount();

        log.info("[START] Bắt đầu đặt đơn hàng. OrderId: {}, CustomerId: {}, Amount: {}", orderId, customerId, amount);

        try {
            // 1. Kiểm tra kho và Khấu trừ tồn kho tạm thời trong Transaction
            log.info("[INVENTORY] Kiểm tra kho cho đơn hàng. OrderId: {}", orderId);
            deductInventory(order.getItems(), orderId, customerId);

            // 2. Gọi cổng thanh toán của bên thứ ba
            log.info("[PAYMENT] Gọi payment gateway xử lý giao dịch. OrderId: {}", orderId);
            processPayment(order, orderId, customerId, amount);

            // 3. Lưu đơn hàng vào database sau khi thanh toán thành công
            log.info("[SAVE] Lưu đơn hàng vào database trạng thái thành công. OrderId: {}", orderId);
            order.setStatus("SUCCESS");
            Order savedOrder = orderRepository.save(order);

            log.info("[COMPLETED] Hoàn tất xử lý đơn hàng thành công. OrderId: {}", orderId);

            // Trả về DTO kết quả thay vì void
            return OrderPlacementResult.builder()
                    .orderId(savedOrder.getId())
                    .status(savedOrder.getStatus())
                    .totalAmount(savedOrder.getTotalAmount())
                    .createdAt(LocalDateTime.now())
                    .build();

        } catch (OutOfStockException e) {
            log.error("[ERROR] Hết hàng khi xử lý đơn hàng. OrderId: {}. Lý do: {}", orderId, e.getMessage());
            throw e;
        } catch (PaymentFailedException e) {
            log.error("[ERROR] Payment fail - Thanh toán thất bại. OrderId: {}. Lý do: {}", orderId, e.getMessage());
            throw e;
        } catch (Exception e) {
            log.error("[ERROR] Lỗi hệ thống không xác định khi đặt hàng. OrderId: {}. Chi tiết: {}", orderId, e.getMessage(), e);
            throw new OrderException("Hệ thống gặp sự cố không mong muốn khi xử lý đơn hàng: " + orderId, e);
        }
    }

    private void validateOrderStructure(Order order) {
        if (order == null) {
            throw new InvalidOrderException("Dữ liệu đơn hàng đầu vào bị trống (null).");
        }
        if (order.getItems() == null || order.getItems().isEmpty()) {
            throw new InvalidOrderException("Đơn hàng không hợp lệ do không chứa sản phẩm nào.");
        }
    }

    private void deductInventory(List<OrderItem> items, String orderId, String customerId) {
        for (OrderItem item : items) {
            String productId = item.getProductId();
            int requestedQty = item.getQuantity();

            Product product = productRepository.findById(productId)
                    .orElseThrow(() -> new ProductNotFoundException("Không tìm thấy sản phẩm trên hệ thống: " + productId));

            int currentStock = product.getStock();
            if (currentStock <= 0 || requestedQty > currentStock) {
                throw new OutOfStockException(
                        String.format("Sản phẩm %s không đủ tồn kho (Yêu cầu: %d, Hiện có: %d)", 
                                product.getName(), requestedQty, currentStock)
                );
            }

            log.info("[INVENTORY_DEDUCT] Trừ kho sản phẩm: {} | Số lượng: {}. OrderId: {}", productId, requestedQty, orderId);
            product.setStock(currentStock - requestedQty);
            productRepository.save(product);
        }
    }

    private void processPayment(Order order, String orderId, String customerId, BigDecimal amount) {
        try {
            paymentGateway.charge(customerId, amount);
        } catch (Exception e) {
            throw new PaymentFailedException("Cổng thanh toán phản hồi lỗi hoặc timeout.", e);
        }
    }
}

package com.speedycart.orderservice.service;

import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)
class OrderPlacementServiceTest {

    @Mock
    private ProductRepository productRepository;

    @Mock
    private OrderRepository orderRepository;

    @Mock
    private PaymentGateway paymentGateway;

    @InjectMocks
    private OrderPlacementService orderPlacementService;

    private Order mockOrder;
    private Product mockProduct;

    @BeforeEach
    void setUp() {
        // Khởi tạo mock dữ liệu sản phẩm
        mockProduct = new Product();
        mockProduct.setId("PROD-001");
        mockProduct.setName("Sản phẩm thử nghiệm");
        mockProduct.setStock(10);

        // Khởi tạo mock dữ liệu item đơn hàng
        OrderItem mockItem = new OrderItem("PROD-001", 2);

        // Khởi tạo mock dữ liệu đơn hàng
        mockOrder = new Order();
        mockOrder.setId("ORD-999");
        mockOrder.setCustomerId("CUST-123");
        mockOrder.setTotalAmount(new BigDecimal("500000"));
        mockOrder.setItems(List.of(mockItem));
    }

    @Test
    @DisplayName("Ném PaymentFailedException và KHÔNG lưu đơn hàng khi cổng thanh toán sập ngắt quãng")
    void should_ThrowPaymentFailedException_And_NeverSaveOrder_When_PaymentGatewayThrowsRuntimeException() throws Exception {
        // Given (Thiết lập hành vi giả lập cho các dependency)
        when(productRepository.findById("PROD-001")).thenReturn(Optional.of(mockProduct));
        
        // Mô phỏng kịch bản cổng thanh toán ném ra lỗi Runtime hệ thống
        doThrow(new RuntimeException("Kết nối cổng thanh toán bị ngắt đột ngột!"))
                .when(paymentGateway).charge(eq("CUST-123"), any(BigDecimal.class));

        // When & Then (Thực thi hành vi và kiểm tra ngoại lệ)
        assertThrows(PaymentFailedException.class, () -> {
            orderPlacementService.placeOrder(mockOrder);
        }, "Kỳ vọng hệ thống bọc lỗi và ném ra PaymentFailedException.");

        // Verify (Xác thực hành vi của Mockito)
        
        // 1. Xác thực phương thức thanh toán thực sự đã được gọi để xử lý giao dịch
        verify(paymentGateway, times(1)).charge(eq("CUST-123"), any(BigDecimal.class));
        
        // 2. Điểm mấu chốt: Xác thực hành vi lưu đơn hàng KHÔNG bao giờ được thực hiện do đã dính lỗi trước đó
        verify(orderRepository, never()).save(any(Order.class));
        
        // 3. Xác thực tầng Repository của sản phẩm có được gọi (để chạy qua luồng trừ kho trước khi sập thanh toán)
        verify(productRepository, times(1)).save(any(Product.class));
    }
